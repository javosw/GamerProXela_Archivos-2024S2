GamerProXela:DATABASE {
	Administracion:SCHEMA {
		Sucursales:TABLE {
			(PK) id_sucursal:string
			nombre:string
			ubicacion:string
		}
		Empleados:TABLE {
			(PK) dpi:number
			nombre:string
			Sucursales.id_sucursal
			rol:enum[caja,bodega,inventario,administracion]
			(UK) username:string
			password:string
		}
	}
	Bodega:SCHEMA {
		Productos:TABLE {
			(PK) id_producto:barcode
			nombre:string
		}
		ProductosSucursal:TABLE {
			(PK) Sucursales.id_sucursal
			(PK) Productos.id_producto
			unidades_en_bodega:number
			unidades_en_estanteria:number
		}
	}
	Inventario:SCHEMA {
		Estanteria:TABLE {
			(PK) Sucursales.id_sucursal
			(PK) id_pasillo:number
			(PK) Productos.id_producto
			unidades_en_pasillo:number
		}
	}
	Caja:SCHEMA {
		Cajas:TABLE {
			(PK) id_caja:number
			(PK) Sucursales.id_sucursal
			Empleados.dpi
		}
		Clientes:TABLE {
			(PK) nit:number
			nombre:string
			total_historico:number
		}
		Ventas:TABLE {
			(PK) id_factura:number
			Caja.id_caja
			Sucursales.id_sucursal
			Clientes.nit
			total:number
			total_descuento:number
			fecha:date
		}
		ProductosFacturados:TABLE {
			(PK) Ventas.id_factura
			(PK) Productos.id_producto
			unidades:number
			subtotal:number
		}
		Tarjetas:TABLE {
			(PK) Clientes.nit
			nivel:enum[comun,oro,platino,diamante]
			puntos:number
		}
	}
}
